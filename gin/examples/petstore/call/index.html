GET /pets

üöÄ Code Comparison

Before (Traditional Handler)

doc.Post("/pets").
    Summary("Create a pet").
    RequestBody("application/json", arrest.ModelFrom[CreatePetRequest]()).
    Response("200", func(r *arrest.Response) {
        r.Description("Success").
            Content("application/json", arrest.ModelFrom[Pet]())
    }).
    Handler(func(c *gin.Context) {
        var req CreatePetRequest
        if err := c.ShouldBindJSON(&req); err != nil {
            c.JSON(400, gin.H{"error": err.Error()})
            return
        }

        pet, err := CreatePet(c.Request.Context(), req)
        if err != nil {
            c.JSON(500, gin.H{"error": err.Error()})
            return
        }

        c.JSON(200, pet)
    })

After (New Call Method)

doc.Post("/pets").
    Summary("Create a pet").
    Call(CreatePet)

That's it! The Call method automatically handles all the boilerplate for you.
